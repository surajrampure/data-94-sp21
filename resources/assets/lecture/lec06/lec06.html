def fancier_print(n):
    if n == 23:
        print('I love this number!')
    else:
        print('This is not my favorite number.')
    print(n)

In [10]:

You don't always need to use else.

In [11]:
def safe_divide(a, b):
    if b == 0:
        return 0
    return a / b
In [12]:
safe_divide(5, 0)
Out[12]:
0
In [13]:
safe_divide(5, 2)
Out[13]:
2.5

The above is equivalent to this; I personally prefer the above.

def safe_divide(a, b):
    if b == 0:
        return 0
    else:
        return a / b

def state_color(state, vote):
    if state == 'California':
        return 'green'
    elif vote == 'D':
        return 'blue'
    else:
        return 'red'
